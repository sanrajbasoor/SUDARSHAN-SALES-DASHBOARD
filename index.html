// ───────────────────────────────────────────────────────────── // WEB APP HANDLERS // ───────────────────────────────────────────────────────────── function doGet(e) { return getPayload(); } function doPost(e) { try { return savePayload(e.postData.contents); } catch(err) { return respond({ error: err.message }, false); } } function getPayload() { const sheet = getOrCreateSheet(); const raw = sheet.getRange('A1').getValue(); const output = ContentService.createTextOutput(raw || '{"enquiries":[],"config":{}}'); output.setMimeType(ContentService.MimeType.JSON); return output; } function savePayload(rawJson) { const sheet = getOrCreateSheet(); sheet.getRange('A1').setValue(rawJson); return respond({ ok: true, savedAt: new Date().toISOString() }, true); } function respond(obj, success) { const out = ContentService.createTextOutput(JSON.stringify(Object.assign({ success }, obj))); out.setMimeType(ContentService.MimeType.JSON); return out; } function getOrCreateSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.getRange('A1').setValue('{"enquiries":[],"config":{}}'); sheet.setColumnWidth(1, 400); } return sheet; } // ───────────────────────────────────────────────────────────── // MORNING DIGEST — runs daily at 8 AM via time trigger // ───────────────────────────────────────────────────────────── function sendMorningDigest() { const data = loadData(); if (!data) return; const { enquiries, config } = data; const today = todayStr(); const todayD = new Date(today); const month = today.slice(0, 7); // ── Compute stats ────────────────────────────────────────── const ACTIVE = ['New','Contacted','Qualified','Test Drive','Negotiation']; const STALE_DAYS = 7; const activeEnquiries = enquiries.filter(e => ACTIVE.includes(e.stage)); const todayFollowUps = activeEnquiries.filter(e => e.nextFollowUp === today); const overdueFollowUps = activeEnquiries.filter(e => e.nextFollowUp && e.nextFollowUp < today); const stale = activeEnquiries.filter(e => { if (!e.lastContactAt) return true; const days = Math.round((todayD - new Date(e.lastContactAt)) / 86400000); return days > STALE_DAYS; }); const thisMonthEnq = enquiries.filter(e => (e.createdAt||'').slice(0,7) === month); const monthWon = thisMonthEnq.filter(e => e.stage === 'Won').length; const monthNew = thisMonthEnq.length; // Yesterday's new enquiries const yesterday = new Date(todayD); yesterday.setDate(yesterday.getDate()-1); const yStr = yesterday.toISOString().slice(0,10); const yesterdayNew = enquiries.filter(e => e.createdAt === yStr).length; // Pacing const dayOfMonth = todayD.getDate(); const daysInMonth = new Date(todayD.getFullYear(), todayD.getMonth()+1, 0).getDate(); const globalTarget = (config.targets && config.targets.global) || 0; const expectedNow = globalTarget ? Math.round(globalTarget * dayOfMonth / daysInMonth) : 0; const daysLeft = daysInMonth - dayOfMonth; const needed = Math.max(0, globalTarget - monthWon); // Counter breakdown of today's follow-ups const counters = (config.counters || []); const counterFU = counters.map(c => { const n = todayFollowUps.filter(e => e.counter === c).length; return n > 0 ? `${c}: ${n}` : null; }).filter(Boolean); // Stale by rep const reps = (config.reps || []); const staleByRep = reps.map(r => { const n = stale.filter(e => e.rep === r).length; return n > 0 ? `${r}: ${n}` : null; }).filter(Boolean); // ── Build email body ─────────────────────────────────────── const now = new Date(); const dateStr = now.toLocaleDateString('en-IN', { weekday:'long', day:'numeric', month:'long', year:'numeric' }); const paceStatus = !globalTarget ? 'No target set' : monthWon >= expectedNow ? `✅ On track (${monthWon}/${globalTarget}, needed ${expectedNow})` : `⚠️ Behind pace (${monthWon}/${globalTarget}, needed ${expectedNow})`; // WhatsApp-friendly format (emoji + clean text) const waText = [ `🏍️ *${DEALER_NAME} — Morning Digest*`, `📅 ${dateStr}`, ``, `━━━ 📊 THIS MONTH (${formatMonth(month)}) ━━━`, `🎯 Target: *${globalTarget || '—'}* bookings`, `✅ Won so far: *${monthWon}* | Expected today: *${expectedNow}*`, `📌 Still needed: *${needed}* in ${daysLeft} days`, `📈 Status: ${paceStatus}`, ``, `━━━ 📋 TODAY'S FOLLOW-UPS ━━━`, `🔔 Due today: *${todayFollowUps.length}*${counterFU.length ? ' (' + counterFU.join(', ') + ')' : ''}`, `🚨 Overdue: *${overdueFollowUps.length}*`, overdueFollowUps.length > 0 ? overdueFollowUps.slice(0,5).map(e => ` • ${e.customer} (${e.counter}) — ${e.rep}`).join('\n') : '', ``, `━━━ ⚠️ STALE LEADS ━━━`, `No contact > 7 days: *${stale.length}*`, staleByRep.length > 0 ? `By rep: ${staleByRep.join(' | ')}` : '', stale.length > 0 ? stale.slice(0,5).map(e => ` • ${e.customer} (${e.counter}) — ${e.rep}`).join('\n') : '', ``, `━━━ 📥 YESTERDAY ━━━`, `New enquiries logged: *${yesterdayNew}*`, `Total pipeline (all time): *${monthNew}* this month`, ``, `🔗 Dashboard: https://sanrajbasoor.github.io/SUDARSHAN-SALES-DASHBOARD/`, ].filter(l => l !== null).join('\n'); // HTML email version const htmlBody = `

🏍️ ${DEALER_NAME} — Morning Digest

${dateStr}

📊 ${formatMonth(month)} Target Pacing

${monthWon}
Bookings won
${expectedNow}
Expected today
${globalTarget||'—'}
Monthly target
${needed}
Still needed (${daysLeft} days)

📋 Today's Follow-ups

Due today: ${todayFollowUps.length}${counterFU.length ? ' — ' + counterFU.join(', ') : ''}

Overdue: ${overdueFollowUps.length}

${overdueFollowUps.slice(0,5).map(e => `
• ${e.customer} · ${e.counter} · ${e.rep} (due ${e.nextFollowUp})
`).join('')}

⚠️ Stale Leads (no contact > 7 days)

${stale.length} leads at risk${staleByRep.length ? ' — ' + staleByRep.join(' | ') : ''}

${stale.slice(0,5).map(e => `
• ${e.customer} · ${e.counter} · ${e.rep}
`).join('')}

📥 Yesterday

New enquiries logged: ${yesterdayNew}

Open Dashboard →
`; // Send email try { MailApp.sendEmail({ to: OWNER_EMAIL, subject: `🏍️ ${DEALER_NAME} Morning Digest — ${dateStr}`, htmlBody: htmlBody, body: waText // plain text fallback }); console.log('Morning digest sent to ' + OWNER_EMAIL); } catch(err) { console.error('Email send failed: ' + err.message); } // Also save digest text to a "Digests" sheet for reference try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let digestSheet = ss.getSheetByName('Digests'); if (!digestSheet) { digestSheet = ss.insertSheet('Digests'); digestSheet.appendRow(['Date','Won','Expected','Target','Stale','Follow-ups Due','Overdue']); } digestSheet.appendRow([today, monthWon, expectedNow, globalTarget, stale.length, todayFollowUps.length, overdueFollowUps.length]); } catch(err) { console.warn('Could not log digest: ' + err.message); } } // ───────────────────────────────────────────────────────────── // TRIGGER SETUP — run once manually to create the daily trigger // ───────────────────────────────────────────────────────────── function setupDailyDigestTrigger() { // Delete existing digest triggers first ScriptApp.getProjectTriggers().forEach(t => { if (t.getHandlerFunction() === 'sendMorningDigest') { ScriptApp.deleteTrigger(t); } }); // Create new trigger: 8–9 AM every day ScriptApp.newTrigger('sendMorningDigest') .timeBased() .everyDays(1) .atHour(8) .create(); console.log('✅ Daily digest trigger created — will run at 8 AM every day.'); } // ───────────────────────────────────────────────────────────── // HELPERS // ───────────────────────────────────────────────────────────── function loadData() { try { const sheet = getOrCreateSheet(); const raw = sheet.getRange('A1').getValue(); if (!raw) return null; return JSON.parse(raw); } catch(err) { console.error('loadData failed: ' + err.message); return null; } } function todayStr() { return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd'); } function formatMonth(ym) { const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec']; const [y, m] = ym.split('-'); return months[parseInt(m,10)-1] + ' ' + y; }
